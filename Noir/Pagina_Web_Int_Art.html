// Ejemplo de código en Python para procesamiento de lenguaje natural

                
import nltk
from nltk.tokenize import word_tokenize

text = "La inteligencia artificial es fascinante."
tokens = word_tokenize(text)

print("Palabras en el texto:")
for token in tokens:
    print(token)
